# 题目
反转一个单链表。
```javascript
输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL
```

# 解题
[参考链表基本数据结构](./链表基本数据结构.md)
自己实现的
```javascript
function reverseList (headNode) {
  let cur = headNode.head
  let pre = null
  let next = null
  while (cur) {
    // debugger
    next = cur.next
    cur.next = pre
    pre =  cur
    cur = next
  }
  headNode.head = pre
  return headNode
}
// stop()
const linked = new LinkedList([1,2,3,4,5])
console.log(reverseList(linked).print())
```

var reverseList = function(head) {
    let pre = null
    let cur = head
    while(cur) {
        let next = cur.next
        cur.next = pre
        pre = cur
        cur = next
    }
    return pre
};
